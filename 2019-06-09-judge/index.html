判断策略

June 2019 · 2 minute read

if-else

function checkStatus(status) {
  if (status === 'start') {
    console.log('开始')
  } else if (status === 'loading') {
    console.log('加载中...')
  } else if (status === 'end') {
    console.log('结束')
  } else {
    console.log('网络异常')
  }
}

switch

function checkStatus(status) {
  switch(status) {
    case 'start':
      console.log('开始')
      break;
    case 'loading':
      console.log('加载中...')
      break;
    case 'end':
      console.log('结束')
      break;
    default:
      console.log('网络异常')
  }
}

Object 判断

let mode = {
  'start': ['开始'],
  'loading': ['加载中...'],
  'end': ['结束'],
  'default': ['网络异常']
}

const checkStatus = (status) => {
  let way = mode[status] || mode['default']
  console.log(way)
}

checkStatus('loading') // console.log('加载中...')

Map 判断

const mode = new Map([
  ['start', ['开始']],
  ['loading', ['加载中...']],
  ['end', ['结束']],
  ['default', ['网络异常']]
])

const checkStatus = (status) => {
  let way = mode.get(status) || mode.get('default')

  console.log(way)
}


checkStatus('loading') // ["加载中..."]

Object

let mode = {
  'wifi_start': () => { console.log('WiFi 环境下，开始') },
  'wifi_loading': () => { console.log('WiFi 环境下，加载中...') },
  'wifi_end': () => { console.log('WiFi 环境下，结束') },
  '4G_start': () => { console.log('4G网络环境下，开始') },
  '4G_loading': () => { console.log('4G网络环境下，加载中...') },
  '4G_end': () => { console.log('4G网络环境下，结束') },
  'default': () => { console.log('网络异常') }
}

const checkNet = (network, status) => {
  let net = mode[`${network}_${status}`] || mode['default']
  net.call(this)
}

/**
 * 检测网络环境，并返回应用状态 
 */
checkNet('wifi', 'end') // WiFi 环境下，结束
